SELECT rekomendasi, jenisHomestay,nama_properti,aksesibilitas,deskripsi,
                                    hal_sekitar,tanggalCheckin,ranjang, alamat 
                                    FROM deskripsi JOIN profil USING(id_profil)
				    JOIN homestay USING(id_homestay);
